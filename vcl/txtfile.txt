#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const ll INF = 1e18;
const int MOD = 1e9 + 7;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int n;
    cin>>n;
    int sum = (n*(n+1))/2;
    vector<vector<int>> dp(n+1, vector<int> (sum+1, 0));
    for(int i = 0; i <= n; i++) dp[i][0] = 1;
    for(int i = 1; i <= n; i++) {
        for(int j = 1; j <= sum; j++) {
            if(i<=j) {
                dp[i][j] = (dp[i-1][j]+dp[i-1][j-i])%MOD;
            } else dp[i][j] = dp[i-1][j];
            cout<<dp[i][j]<<" ";
        }
        cout<<'\n';
    }
    int ans = 0;
    for(int i = 1; i <= sum; i++) {
        if(dp[n][i]>1) {
            int x = dp[n][i];
            ans += ((x*(x-1))/2)/MOD;
        }
    }
    cout<<ans<<'\n';
    return 0;
}
